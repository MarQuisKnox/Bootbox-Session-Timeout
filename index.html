Bootbox Session Timeout Demo

<script type="text/javascript">
    $(document).ready(function() {
        bootbox.alert('This is a demo of Bootbox Session Timeout');

        $(document).idleTimeout({ 
            inactivity: 30000, 
            noconfirm: 10000, 
            sessionAlive: 10000 
        });
    }); 
</script>
